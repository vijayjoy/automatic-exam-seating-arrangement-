CREATE TABLE IF NOT EXISTS admins (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    email TEXT UNIQUE NOT NULL,
    password TEXT NOT NULL
  );
CREATE TABLE IF NOT EXISTS students (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    register_number TEXT UNIQUE NOT NULL,
    name TEXT DEFAULT 'Student',
    dob TEXT DEFAULT '01/01/2000',
    course TEXT NOT NULL,
    year TEXT NOT NULL
  );
CREATE TABLE IF NOT EXISTS seats (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    classroom_number INTEGER NOT NULL,
    bench_number INTEGER NOT NULL,
    seat_position TEXT NOT NULL, /* L for left, R for right */
    student_id INTEGER,
    FOREIGN KEY(student_id) REFERENCES students(id)
  );
CREATE TABLE IF NOT EXISTS classrooms (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    custom_id TEXT,
    name TEXT NOT NULL UNIQUE,
    benches INTEGER NOT NULL DEFAULT 35,
    capacity_per_bench INTEGER NOT NULL DEFAULT 2,
    teachers TEXT,
    required_invigilators INTEGER NOT NULL DEFAULT 2
  );
CREATE TABLE IF NOT EXISTS exam_selections (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    exam_date TEXT NOT NULL,
    course_name TEXT NOT NULL,
    total_students INTEGER NOT NULL,
    generated_by TEXT NOT NULL
  );
CREATE TABLE IF NOT EXISTS teachers (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    name TEXT NOT NULL,
    department TEXT,
    status TEXT DEFAULT 'Available',
    assigned_room_id INTEGER DEFAULT NULL,
    FOREIGN KEY(assigned_room_id) REFERENCES classrooms(id)
  );
CREATE TABLE IF NOT EXISTS admin_logs (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    action TEXT NOT NULL,
    timestamp DATETIME DEFAULT CURRENT_TIMESTAMP
  );
CREATE TABLE IF NOT EXISTS courses (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    name TEXT NOT NULL UNIQUE,
    code TEXT UNIQUE,
    department TEXT,
    description TEXT
  );
